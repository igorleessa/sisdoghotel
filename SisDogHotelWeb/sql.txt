create table funcionario(
idfuncionario int identity(1,1) not null,
nome nvarchar(50) not null,
loginfuncionario nvarchar(20) not null,
senhafuncionario nvarchar(50) not null,
nivelacesso nvarchar(30) not null,
primary key(idfuncionario)
)

create table animal(
idanimal int identity(1,1) not null,
nome nvarchar(50) not null,
alimentacao nvarchar(50) not null,
cor nvarchar(50) not null,
idfuncionario int not null,
primary key(idanimal),
foreign key(idfuncionario) references funcionario(idfuncionario)
)


create table rotinaalimentacao(
idrotinaalimentacao int identity(1,1) not null,
quantidade decimal not null,
horarioalimentacao datetime not null,
data datetime not null,
idfuncionario int not null,
idanimal int not null,
primary key(idrotinaalimentacao),
foreign key(idfuncionario) references funcionario(idfuncionario),
foreign key(idanimal) references animal(idanimal)
)



alter table animal drop column alimentacao
alter table animal add observacao nvarchar(500)

create table alimentacao(
idalimentacao int identity(1,1) not null,
nomealimentacao nvarchar(50) not null,
tipo nvarchar(50) not null,
valor nvarchar(50) not null,
descricao nvarchar(500) not null
primary key(idalimentacao)
)

alter table rotinaalimentacao add idalimentacao int not null, foreign key(idalimentacao) references alimentacao(idalimentacao)

create table cliente(
idcliente int identity(1,1) not null,
nome nvarchar(50) not null,
sobrenome nvarchar(50) not null,
datanascimento datetime not null,
telefone nvarchar(20) not null,
celular nvarchar(20) not null,
cep int not null,
rua nvarchar(50) not null,
numerocasa nvarchar(20) not null,
complemento nvarchar(50) not null,
cidade nvarchar(50) not null,
estado nvarchar(50) not null,
pais nvarchar(50) not null,
cpf int not null,
rg int not null,
usuariocliente nvarchar(20) not null,
senhacliente nvarchar(50) not null,
primary key(idcliente)
)

alter table animal add idcliente int not null, foreign key(idcliente) references cliente(idcliente)

create table reserva(
idreserva int identity(1,1) not null,
valor decimal not null,
sinal decimal not null,
datadareserva datetime not null,
datafinal datetime,
primary key(idreserva)
)
create table quarto(
idquarto int identity(1,1) not null,
numeroquarto int not null,
descricao nvarchar(500) not null,
valor decimal not null,
primary key(idquarto),
)

create table checkin(
idcheckin int identity(1,1) not null,
datainicio datetime not null,
statuscheckin nvarchar(50) not null,
primary key(idcheckin)
)

create table checkout(
idcheckout int not null,
statuscheckout nvarchar(50) not null,
custofinal decimal not null,
custoadicional decimal,
primary key(idcheckout)
)

create table pagamento(
idpagamento int identity(1,1) not null,
valortotal decimal not null,
tipopagamento nvarchar(20) not null,
primary key(idpagamento)
)
alter table pagamento add idcheckout int not null, idreserva int not null, foreign key(idcheckout) references checkout(idcheckout), foreign key(idreserva) references reserva(idreserva)
